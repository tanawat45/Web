CREATE TABLE events(
    event_id int UNSIGNED NOT NULL AUTO_INCREMENT,
    name varchar(60) NOT NULL,  
    img varchar(60),
    schedule_event varchar(100),
    rules varchar(100),
    charge varchar(100),
    map varchar(100),
    generation_competing varchar(60),
    race_day DATE,
    flag_full int(1),
    num_full int,
    charge_full int(4) ,
    flag_half int(1),
    num_half int,
    charge_half int(4) ,
    flag_mini int(1),
    num_mini int,
    charge_mini  int(4) ,
    flag_fun int(1),
    num_fun int,
    charge_fun int(4) ,
    phone varchar(10),
    email varchar(30),
    shirtimg1 varchar(60),
    shirtimg2 varchar(60),
    PRIMARY KEY(event_id)
)
